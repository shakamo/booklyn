def self.multi_trim(value)

  trim_value = String.new(value)

  TRIM_STR.each do |str|
    trim_value.gsub!(str, '')
  end

  trim_value.gsub!(' ','')
  trim_value.gsub!('　','')
  return trim_value
end

def self.query_trim(value)

  trim_value = String.new(value)

  TRIM_STR.each do |str|
    trim_value.gsub!(str, '%')
  end

  trim_value.gsub!(' ','%')
  trim_value.gsub!('　','%')
  trim_value.gsub!('%%','%')
  return trim_value
end

def self.regex_trim(value)
  trim_value = query_trim(value)

  trim_value.gsub!('%','.*')
  return trim_value
end
